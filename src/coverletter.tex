% Cover-letter specific typography aligned with resume theme

% How the letter is opened
\letteropening{Dear Hiring Manager,}
% How the letter is closed
\letterclosing{Sincerely,}
% Any enclosures with the letter
\letterenclosure[Attached]{Resume}


%-------------------------------------------------------------------------------
\begin{document}

% Print the header with above personal information
% Give optional argument to change alignment(C: center, L: left, R: right)

% Leave any of these blank if they are not needed
\makecvfooter

% Print the title with above letter information
\makelettertitle

%-------------------------------------------------------------------------------
%	LETTER CONTENT
%-------------------------------------------------------------------------------
\begin{cvletter}

  \lettersection{Opening}
  I am writing to apply for the <Role Title> position at <Company Name>. I am a Front-End Engineer focused on building high-quality, scalable product experiences with React, Next.js, and TypeScript. My recent work combines delivery speed, UI quality, and reliability across complex, data-heavy applications.

  \lettersection{Relevant Impact}
  At <Most Recent Company>, I led initiatives that improved release confidence and engineering throughput, including stronger test coverage, component system improvements, and CI/CD optimizations. This resulted in fewer production issues, faster build/deploy cycles, and more predictable delivery for product teams. I also partnered closely with design, backend, and product stakeholders to ship features with strong performance and maintainability.

  \lettersection{Why <Company Name>}
  I am particularly interested in <Company Name> because of <product/mission/market focus>. I can contribute quickly by delivering accessible, reliable interfaces, strengthening frontend architecture, and helping teams ship faster with better quality controls. I would welcome the opportunity to discuss how my experience aligns with your goals for this role.

\end{cvletter}


%-------------------------------------------------------------------------------
% Print the signature and enclosures with above letter information
\makeletterclosing

\end{document}
